//01背包问题<!-- more -->
```
#include <iostream>
using namespace std;
int main ()
{
    int v,n,a[31],dp[20001]={0};
    cin>>v>>n;
    for(int i=1; i<=n; i++){
        cin>>a[i];
    }
    for(int j=v; j>=a[i]; j--){ //从a[i]起就是完全背包问题
        dp[j]=max(dp[j], dp[j-a[i]]+a[i]);
    }    
    for(int j=0; j<=v; j++){
        cout<<dp[j]<<' ';
    }    
    cout<<endl;
    }
    cout<<v-dp[v]<<endl;
}
```
